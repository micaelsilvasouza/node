Node bcrypt

Codificar valores em hash

        npm install bcryptjs

Cria um hash apartir de um salt, que gerar mais complexidade ao hash

        const bcrypt = require("bcrypt")
        var senha

        bcrypt.genSalt(2, (err, salt)=>{
            bcrypt.hash("testando", salt, (err, hash)=>{
                senha = hash
            })
        })

Para comparar um valor ao um hash, use o metodo compare

        bcrypt.compare("testando", senha, (err, batem)=>{
            console.log(batem)
        })
